Find Maximum Sum Path in Matrix using DP

#include <stdio.h>

void findMax(int fsmax[], int r, int n, int dp[][n]) {
    int fmax = -1, smax = -1;
    for (int j = 0; j < n; j++) {
        if (dp[r][j] > fmax) {
            smax = fmax;
            fmax = dp[r][j];
        } else if (dp[r][j] > smax) {
            smax = dp[r][j];
        }
    }
    fsmax[0] = fmax;   
    fsmax[1] = smax;   
}

int main() {
    int m, n;
    scanf("%d%d", &m, &n);
    int a[m][n], dp[m][n];
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    for (int j = 0; j < n; j++) {
        dp[0][j] = a[0][j];
    }

    int fsmax[2]; 
    for (int i = 1; i < m; i++) {
        findMax(fsmax, i - 1, n, dp);
        for (int j = 0; j < n; j++) {
            if (j == fsmax[2]) {
                dp[i][j] = a[i][j] + fsmax[1]; 
            } else {
                dp[i][j] = a[i][j] + fsmax[0]; 
            }
        }
    }
    int ans = -1;
    for (int j = 0; j < n; j++) {
        if (dp[m - 1][j] > ans) ans = dp[m - 1][j];
    }

    printf("%d\n", ans);
    return 0;
}
