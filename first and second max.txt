FIRST AND SECOND MAX

#include<stdio.h>
int main()
{
    int n,fimax,curr,smax=-1;
    scanf("%d%d",&n,&fimax);
    for(int i=2;i<=n;i++)
    {
        scanf("%d",&curr);
        if(curr>fimax)
        {
            smax=fimax;
            fimax=curr;
        }
        else if(curr>smax)
        {
            smax=curr;
        }
    }
    printf("%d %d",fimax,smax);
}
